        { h: 20, task: "📚 পড়াশোনা ৩", type: "study", label: "08 PM - 09 PM" },
        { h: 21, task: "📚 পড়াশোনা ৩", type: "study", label: "09 PM - 10 PM" },
        { h: 22, task: "📚 পড়াশোনা ৩", type: "study", label: "10 PM - 11 PM" },
        { h: 23, task: "🍲 খাবার/বিশ্রাম", type: "other", label: "11 PM - 12 AM" },
        { h: 0, task: "📚 পড়াশোনা ৪", type: "study", label: "12 AM - 01 AM" }
    ];

    let studyLog = new Set();

    function renderDiary() {
        const board = document.getElementById('diaryBoard');
        const now = new Date();
        document.getElementById('dateText').innerText = now.toLocaleDateString('bn-BD', { weekday: 'long', day: 'numeric', month: 'long', year: 'numeric' });

        scheduleData.forEach(item => {
            const row = document.createElement('div');
            row.className = `slot border-${item.type}`;
            row.id = `row-${item.h}`;
            
            row.innerHTML = `
                <div class="slot-top">
                    <span class="time-label">${item.label}</span>
                    <span class="task-type-label">${item.task}</span>
                </div>
                <input type="text" class="plan-input" placeholder="এখানে তোমার আজকের পরিকল্পনা লেখো...">
                <div class="action-area">
                    <span style="font-size: 0.75rem; color: #777;">সম্পন্ন হয়েছে?</span>
                    <div class="status-btns">
                        <button class="btn btn-check" onclick="updateStatus('${item.h}', true)">✓</button>
                        <button class="btn btn-cross" onclick="updateStatus('${item.h}', false)">✕</button>
                    </div>
                </div>
            `;
            board.appendChild(row);
        });
    }

    function updateStatus(h, isDone) {
        const row = document.getElementById(`row-${h}`);
        const taskObj = scheduleData.find(x => x.h == h);
        
        row.classList.remove('done', 'failed');
        
        if (isDone) {
            row.classList.add('done');
            if (taskObj.type === 'study') studyLog.add(h);
        } else {
            row.classList.add('failed');
            if (taskObj.type === 'study') studyLog.delete(h);
        }
        
        refreshReport();
    }

    function refreshReport() {
        const count = studyLog.size;
        const totalStudyHours = 10;
        const percentage = (count / totalStudyHours) * 100;
        
        document.getElementById('studyCount').innerText = `${count}/${totalStudyHours} ঘণ্টা`;
        document.getElementById('progressBar').style.width = Math.min(percentage, 100) + "%";
        
        let stars = Math.ceil(count / 2); 
        if (stars > 5) stars = 5;
        document.getElementById('liveStars').innerText = "★".repeat(stars) + "☆".repeat(5 - stars);
    }

    renderDiary();
